country_event = {
    id = 176401
    title = "The Navigation Acts"
    desc = "EVTDESC176401"
    picture = boston_tea
	fire_only_once = yes
	
	trigger = {
		OR = {
			NOT = {
				any_owned_province = {
					has_province_modifier = salutary_neglect
				}
			}
			year = 1740
		}
		any_owned_province = {
			is_core = THC
		}
		OR = { primary_culture = british AND = { primary_culture = scottish NOT = { exists = ENL } } }
	}

    option = {
        name = "Those colonials will get over it."
        THC = {
			all_core = {
				add_province_modifier = {
					name = american_smugglers
					duration = 5475
				}
				add_province_modifier = {
					name = navigation_acts_enforced
					duration = 7300
				}
			}
		}
		THC = { all_core = { remove_province_modifier = salutary_neglect } }
		set_country_flag = colonists_enraged
		set_variable = {
			which = usa_rev
			value = 2
		}
    }
}
